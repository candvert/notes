使用的是[telescope](https://github.com/nvim-telescope/telescope.nvim)
```lua
require("lazy").setup({
  spec = {
	{
		'nvim-telescope/telescope.nvim',
		dependencies = { 
			'nvim-lua/plenary.nvim',
			'nvim-treesitter/nvim-treesitter'
		},
		-- 设置按键映射，<leader>一般为空格，即查找文件快捷键为 ff
		config = function()
			local builtin = require('telescope.builtin')
			vim.keymap.set('n', '<leader>ff', builtin.find_files, { desc = 'Telescope find files' })
			-- 搜索当前目录下的字符串
			vim.keymap.set('n', '<leader>fg', builtin.live_grep, { desc = 'Telescope live grep' })
			-- 在缓冲区的文件列表中进行搜索
			vim.keymap.set('n', '<leader>fb', builtin.buffers, { desc = 'Telescope buffers' })
			-- 在telescope的帮助手册中进行搜索
			vim.keymap.set('n', '<leader>fh', builtin.help_tags, { desc = 'Telescope help tags' })
		end,
    },
  },
})
```

```
picker非常灵活，支持全局配置和针对单个选择器的配置
```

```lua
require('telescope').setup{
  defaults = {
    -- 在这里填写默认配置:
    -- config_key = value,
    mappings = {
      i = {
        -- 将actions.which_key映射到<C-h>
        -- actions.which_key显示该picker的按键映射
        ["<C-h>"] = "which_key",
		["<esc>"] = require('telescope.actions').close,
		["<C-n>"] = false,
      }
    }
  },
  pickers = {
    -- 内置picker的默认配置写在这里：
    -- picker_name = {
    --   picker_config_key = value,
    --   ...
    -- }
	-- 现在picker_config_key会在你每次调用该内置picker时生效
  },
  extensions = {
    -- 扩展的配置写在这里：
    -- extension_name = {
    --   extension_config_key = value,

  }
}


对于find_files这个picker函数
builtin.find_files({opts})
Options:
{cwd}           搜索的目录
{hidden}        是否显示隐藏文件
设置快捷键 en，即在配置文件目录进行搜索
vim.keymap.set('n', '<space>en', function()
	require('telescope.builtin').find_files {
		cwd = vim.fn.stdpath('config')
	}
end)

使用layout_strategy和layout_config选项设置布局
对于单个picker设置
require('telescope.builtin').find_files({layout_strategy='vertical',layout_config={width=0.5}})
对全局设置
require('telescope').setup({
  defaults = {
    layout_config = {
      vertical = { width = 0.5 }
      -- other layout configuration here
    },
    -- other defaults configuration here
  },
  -- other configuration values here
})
```
